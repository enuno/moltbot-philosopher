## Task 1: Create P2.2 Types Definitions

**Files:**

- Modify: `services/engagement-service/src/types.ts` (add new interfaces)

**Step 1: Add P2.2 interfaces to types.ts**

Append these interfaces to the existing types file:

```typescript
// P2.2: Content Quality Metrics

export interface ThreadDepthMetrics {
  threadId: string;
  maxDepth: number;
  totalComments: number;
  avgBranchingFactor: number;
  uniqueParticipants: number;
  lastActivityAt: string;
}

export interface AuthorThreadEngagement {
  authorId: string;
  authorName: string;
  commentsByAuthor: number;
  repliesReceivedByAuthor: number;
  replyEngagementRate: number;
}

export interface ThreadSentimentMetrics {
  isControversialTopic: boolean;
  controversialTags: string[];
  sentimentSampleSize: number;
  avgSentiment: number;
  sentimentVariance: number;
}

export interface QualityScoreBreakdown {
  depthComponent: number;
  authorEngagementComponent: number;
  sentimentComponent: number;
  depthWeight: number;
  authorEngagementWeight: number;
  sentimentWeight: number;
}

export interface ThreadQualityMetrics {
  threadId: string;
  rootPostId: string;
  depth: ThreadDepthMetrics;
  sentiment: ThreadSentimentMetrics;
  topAuthors: AuthorThreadEngagement[];
  qualityScore: number;
  breakdown: QualityScoreBreakdown;
  computedAt: string;
}

export interface AuthorGlobalMetrics {
  authorId: string;
  totalComments: number;
  totalRepliesReceived: number;
  distinctThreadsParticipated: number;
  avgReplyEngagementRate: number;
  lastUpdatedAt: string;
}

```

Expected: Compilation succeeds with no errors.
