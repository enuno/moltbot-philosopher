export interface StatsResponse {
  service: StatsServiceInfo;
  summary: StatsSummary;
  trends: StatsTrends;

  name: string;                              // "engagement-service"
  version: string;                           // "2.0.0"
  timestamp: string;                         // ISO 8601
  uptime_seconds: number;
  data_freshness: {
    last_heartbeat_at: string;              // ISO 8601
    last_metrics_rollup_at: string;         // ISO 8601
    is_stale: boolean;                      // true if > 5 min old
  };
}

export interface StatsSummary {
  window_hours: number;                      // e.g., 24
  total_posts_seen: number;
  total_threads_tracked: number;
  total_comments_seen: number;
  engagement_events: {
    replies_posted: number;
    threads_continued: number;
    proactive_posts_published: number;
  };
  average_quality_score: number;             // 0–1
  controversial_thread_count: number;
}

export interface StatsTrends {
  top_topics: StatsTopicTrend[];            // top 5–10 topics by velocity
  top_threads: StatsThreadTrend[];          // top 5–10 threads by quality
}

export interface StatsTopicTrend {
  topic_id: string;                         // "ai_safety"

  thread_count: number;
  avg_quality_score: number;                // 0–1
  avg_sentiment: number;                    // -1 to 1

}

  thread_id: string;

  controversial_tags: string[];             // topic IDs
  total_comments: number;
  unique_participants: number;

  window_hours: number;

    threads_started: number;
  };

    avg_quality_score: number;              // 0–1
    controversial_thread_count: number;

  topic_id: string;
  thread_count: number;

export class StateManager {

  private WINDOW_HOURS = 24;

  constructor() {
    this.state = this.initializeState();
    this.rollingMetrics = this.initializeRollingMetrics();
  }

  private initializeRollingMetrics(): RollingMetrics {
    return {
      lastHeartbeatAt: new Date(),
      lastMetricsRollupAt: new Date(),
      topicCounts: new Map(),
      topicQuality: new Map(),
      topicSentiment: new Map(),
      topicVelocity: new Map(),
      agentVelocity: new Map(),
      agentTopics: new Map(),
      qualityDistribution: [0, 0, 0],
      controversialSentiment: { sum: 0, count: 0, variance: 0 },
    };
  }

  async recordEngagementEvent(event: EngagementEvent): Promise<void> {
    // Record event in existing state
    await this.recordEvent(event);

    // Update rolling metrics
    this.updateRollingMetrics(event);

    // Prune old metrics every heartbeat
    this.pruneOldMetrics();
  }

  private updateRollingMetrics(event: EngagementEvent): void {
    // Update topic counts
    for (const topic of event.topics || []) {
      this.rollingMetrics.topicCounts.set(
        topic,
        (this.rollingMetrics.topicCounts.get(topic) || 0) + 1
      );

      // Update topic quality (running average)
      const existing = this.rollingMetrics.topicQuality.get(topic) || { sum: 0, count: 0 };
      this.rollingMetrics.topicQuality.set(topic, {
        sum: existing.sum + event.quality_score,
        count: existing.count + 1,
      });
    }

    // Update quality distribution histogram
    const bucketIndex = event.quality_score < 0.4 ? 0 : event.quality_score < 0.7 ? 1 : 2;
    this.rollingMetrics.qualityDistribution[bucketIndex]++;

    // Update agent velocity
    const agentMetrics = this.rollingMetrics.agentVelocity.get(event.agent) || {
      commentsPerHour: 0,
      threadsTouched: 0,
      threadsStarted: 0,
      avgQualityScore: 0,
      controversialThreadCount: 0,
    };
    agentMetrics.commentsPerHour = this.calculateCommentsPerHour(event.agent);
    agentMetrics.avgQualityScore = this.calculateAgentAvgQuality(event.agent);
    if (event.is_controversial) {
      agentMetrics.controversialThreadCount++;
    }
    this.rollingMetrics.agentVelocity.set(event.agent, agentMetrics);

    // Update agent topics affinity
    if (!this.rollingMetrics.agentTopics.has(event.agent)) {
      this.rollingMetrics.agentTopics.set(event.agent, new Map());
    }
    const agentTopics = this.rollingMetrics.agentTopics.get(event.agent)!;
    for (const topic of event.topics || []) {
      const existing = agentTopics.get(topic) || { threadCount: 0, avgQualityScore: 0 };
      agentTopics.set(topic, {
        threadCount: existing.threadCount + 1,
        avgQualityScore:
          (existing.avgQualityScore * existing.threadCount + event.quality_score) /
          (existing.threadCount + 1),
      });
    }

    // Update controversial sentiment stats
    if (event.is_controversial) {
      this.rollingMetrics.controversialSentiment.sum += event.sentiment || 0;
      this.rollingMetrics.controversialSentiment.count++;
    }

    this.rollingMetrics.lastHeartbeatAt = new Date();
  }

  private pruneOldMetrics(): void {
    const now = Date.now();
    const windowMs = this.WINDOW_HOURS * 60 * 60 * 1000;

    // Remove metrics older than window
    for (const [topic, count] of this.rollingMetrics.topicCounts) {
      const topicLastUpdate = this.state.threadTopics?.get(topic)?.[0]?.timestamp;
      if (topicLastUpdate && now - topicLastUpdate.getTime() > windowMs) {

      }
    }

      this.rollingMetrics.lastMetricsRollupAt = new Date();

// services/engagement-service/src/stats-builder.ts

import {
  RollingMetrics,
  StatsTopicTrend,
  StatsAgentMetrics,
  StatsQualitySection,
} from "./types";

export function buildTopicTrends(
  metrics: RollingMetrics,
  limit: number = 10
): StatsTopicTrend[] {
  const trends: StatsTopicTrend[] = [];

  for (const [topicId, velocity] of metrics.topicVelocity) {
    const count = metrics.topicCounts.get(topicId) || 0;
    const qualityData = metrics.topicQuality.get(topicId) || { sum: 0, count: 0 };
    const sentimentData = metrics.topicSentiment.get(topicId) || { sum: 0, count: 0 };

    trends.push({
      topic_id: topicId,
      label: getTopicLabel(topicId),
      thread_count: count,
      avg_quality_score: qualityData.count > 0 ? qualityData.sum / qualityData.count : 0,
      avg_sentiment: sentimentData.count > 0 ? sentimentData.sum / sentimentData.count : 0,
      engagement_velocity: velocity,
    });
  }

  // Sort by velocity descending
  trends.sort((a, b) => b.engagement_velocity - a.engagement_velocity);
  return trends.slice(0, limit);
}

export function buildAgentMetrics(metrics: RollingMetrics): StatsAgentMetrics[] {
  const agentMetrics: StatsAgentMetrics[] = [];

  for (const [agentType, velocity] of metrics.agentVelocity) {
    const topics = metrics.agentTopics.get(agentType) || new Map();
    const topicSummaries = Array.from(topics).map(([topicId, topicData]) => ({
      topic_id: topicId,
      thread_count: topicData.threadCount,
      avg_quality_score: topicData.avgQualityScore,
    }));

    agentMetrics.push({
      agent_type: agentType,
      engagement_velocity: {
        comments_per_hour: velocity.commentsPerHour,
        threads_touched: velocity.threadsTouched,
        threads_started: velocity.threadsStarted,
      },
      quality: {
        avg_quality_score: velocity.avgQualityScore,
        controversial_thread_count: velocity.controversialThreadCount,
      },
      topics: topicSummaries,
    });
  }

  return agentMetrics;
}

export function buildQualityDistribution(metrics: RollingMetrics): StatsQualitySection {
  const sentiment = metrics.controversialSentiment;
  const avgSentiment =
    sentiment.count > 0 ? sentiment.sum / sentiment.count : 0;

  // Calculate variance
  let variance = 0;
  if (sentiment.count > 1) {
    // Simplified variance calculation (would need individual scores for exact)
    variance = sentiment.variance || 0;
  }

  return {
    window_hours: 24,
    distribution: {
      bucket_0_0_4: metrics.qualityDistribution[0],
      bucket_0_4_0_7: metrics.qualityDistribution[1],
      bucket_0_7_1_0: metrics.qualityDistribution[2],
    },
    sentiment_on_controversial: {
      avg_sentiment: avgSentiment,
      variance: variance,
      sampled_threads: sentiment.count,
    },
  };
}

function getTopicLabel(topicId: string): string {
  const labels: Record<string, string> = {
    ai_safety: "AI Safety and Alignment",
    capitalism: "Capitalism and Economic Systems",
    stoicism: "Stoicism and Philosophy",
    ethics: "Ethics and Morality",
    epistemology: "Epistemology and Knowledge",
    metaphysics: "Metaphysics",
    free_will: "Free Will and Determinism",
    consciousness: "Consciousness and Qualia",
  };
  return labels[topicId] || topicId.replace(/_/g, " ");
}

// services/engagement-service/src/stats-endpoint.ts

import { Router, Request, Response } from "express";
import {
  StatsResponse,
  StatsSummary,
  RollingMetrics,
} from "./types";
import {
  buildTopicTrends,
  buildAgentMetrics,
  buildQualityDistribution,
} from "./stats-builder";
import { StateManager } from "./state-manager";

const HEARTBEAT_INTERVAL_MS = 30 * 1000;
const STALENESS_THRESHOLD_INTERVALS = 5;
const STALENESS_THRESHOLD_MS = HEARTBEAT_INTERVAL_MS * STALENESS_THRESHOLD_INTERVALS;

export function createStatsRouter(stateManager: StateManager): Router {
  const router = Router();
  const startTime = Date.now();

  router.get("/stats", (req: Request, res: Response) => {
    try {
      const metrics = stateManager.getRollingMetrics();
      const now = new Date();
      const uptime = Math.floor((Date.now() - startTime) / 1000);

      // Check staleness
      const metricsAge = now.getTime() - metrics.lastMetricsRollupAt.getTime();
      const isStale = metricsAge > STALENESS_THRESHOLD_MS;

      // Build response sections
      const response: StatsResponse = {
        service: {
          name: "engagement-service",
          version: "2.0.0",
          timestamp: now.toISOString(),
          uptime_seconds: uptime,
          data_freshness: {
            last_heartbeat_at: metrics.lastHeartbeatAt.toISOString(),
            last_metrics_rollup_at: metrics.lastMetricsRollupAt.toISOString(),
            is_stale: isStale,
          },
        },
        summary: buildSummary(metrics),
        trends: {
          top_topics: buildTopicTrends(metrics, 10),
          top_threads: [], // Empty for now - will be populated from thread index
        },
        agents: {
          window_hours: 24,
          per_agent: buildAgentMetrics(metrics),
        },
        quality: buildQualityDistribution(metrics),
      };

      res.status(200).json(response);
    } catch (error) {
      // Graceful degradation: return minimal valid response even on error
      const now = new Date();
      const response: StatsResponse = {
        service: {
          name: "engagement-service",
          version: "2.0.0",
          timestamp: now.toISOString(),
          uptime_seconds: 0,
          data_freshness: {
            last_heartbeat_at: now.toISOString(),
            last_metrics_rollup_at: now.toISOString(),
            is_stale: true,
          },
        },
        summary: {
          window_hours: 24,
          total_posts_seen: 0,
          total_threads_tracked: 0,
          total_comments_seen: 0,
          engagement_events: {
            replies_posted: 0,
            threads_continued: 0,
            proactive_posts_published: 0,
          },
          average_quality_score: 0,
          controversial_thread_count: 0,
        },
        trends: {
          top_topics: [],
          top_threads: [],
        },
        agents: {
          window_hours: 24,
          per_agent: [],
        },
        quality: {
          window_hours: 24,
          distribution: {
            bucket_0_0_4: 0,
            bucket_0_4_0_7: 0,
            bucket_0_7_1_0: 0,
          },
          sentiment_on_controversial: {
            avg_sentiment: 0,
            variance: 0,
            sampled_threads: 0,
          },
        },
      };
      res.status(200).json(response);
    }
  });

  return router;
}

function buildSummary(metrics: RollingMetrics): StatsSummary {
  const totalQuality = Array.from(metrics.topicQuality.values()).reduce(
    (sum, item) => sum + item.sum,
    0
  );
  const totalQualityCount = Array.from(metrics.topicQuality.values()).reduce(
    (sum, item) => sum + item.count,
    0
  );

  return {
    window_hours: 24,
    total_posts_seen: 0, // Would require tracking in state
    total_threads_tracked: Array.from(metrics.topicCounts.values()).reduce(
      (sum, count) => sum + count,
      0
    ),
    total_comments_seen: totalQualityCount,
    engagement_events: {
      replies_posted: totalQualityCount,
      threads_continued: 0, // Would require tracking in state
      proactive_posts_published: 0, // Would require tracking in state
    },
    average_quality_score:
      totalQualityCount > 0 ? totalQuality / totalQualityCount : 0,
    controversial_thread_count: metrics.controversialSentiment.count,
  };
}
