/**

 * Check if a comment text is banned (low-effort)

 * @param text - Comment text to check

 * @param agentType - Agent type to check against agent-specific phrases

 * @returns true if text matches global or agent-specific banned phrases

 */
export function isBannedPhrase(text: string, agentType: string): boolean {
  const lower = text.toLowerCase().trim();

  // Check global banned phrases
  if (GLOBAL_BANNED_PHRASES.some(phrase => lower.includes(phrase.toLowerCase()))) {
    return true;
  }

  // Check agent-specific banned phrases
  const agentBanned = AGENT_BANNED_PHRASES[agentType] || [];
  if (agentBanned.some(phrase => lower.includes(phrase.toLowerCase()))) {
    return true;
  }

  return false;
}

```

/**

 * Calculate engagement velocity score

 * Measures post activity momentum with recency boost

 */
private calculateVelocityScore(post: Post): number {
  const ageMs = Date.now() - new Date(post.createdAt).getTime();
  const commentsPerHour = (post.commentCount / ageMs) * 3600000;

  // Recency boost as specified in P2.1 requirements
  if (ageMs < 3600000) {
    return commentsPerHour * 1.5; // <1h: +50% boost
  }
  if (ageMs < 86400000) {
    return commentsPerHour * 1.2; // <24h: +20% boost
  }
  return commentsPerHour * 1.0;
}

```

/**

 * Calculate feed trend score

 * Checks if post content aligns with platform-wide trending topics

 */
private calculateFeedTrendScore(post: Post): number {
  const trendingTopics = this.trendingTopics || [];
  if (trendingTopics.length === 0) return 0;

  const contentLower = post.content.toLowerCase();

  const matches = trendingTopics.filter(trend =>
    contentLower.includes(trend.topic.toLowerCase())
  ).length;

  if (matches === 0) return 0;

  return Math.min(matches / Math.max(trendingTopics.length, 1), 1.0);
}

// Add property to class

/**

 * Calculate agent interest alignment score

 * Boosts posts matching agent's philosophical interests

 */
private calculateAgentRelevance(post: Post, agentType: string, interests: string[]): number {
  if (!interests || interests.length === 0) return 0.5;

  const contentLower = post.content.toLowerCase();
  const matches = interests.filter(interest =>
    contentLower.includes(interest.toLowerCase())
  ).length;

  const matchScore = Math.min(matches / interests.length, 1.0);

  // Boost strongly matching posts to 1.0, keep others as-is
  return matchScore > 0.7 ? 1.0 : matchScore;
}

 * Calculate trending score from multiple signals

 * Combines: velocity (50%) + feed trends (30%) + agent interests (20%)

 */
calculateTrendingScore(post: Post, agentType: string): number {
  const velocity = this.calculateVelocityScore(post);
  const feedTrend = this.calculateFeedTrendScore(post);
  const agentInterests = this.getAgentInterests(agentType);
  const agentRelevance = this.calculateAgentRelevance(post, agentType, agentInterests);

  return (velocity * 0.5) + (feedTrend * 0.3) + (agentRelevance * 0.2);
}

/**

 * Score post for engagement relevance

 * 5-factor formula: semantic(40%) + trending(30%) + keyword(15%) + author(10%)

 * Plus quality gate: 50% penalty for low-effort comments

 */
async scorePost(post: Post, agent: Agent): Promise<number> {
  // Get base scores from 5 factors
  const semanticScore = await this.getSemanticScore(post, agent);
  const trendingScore = this.calculateTrendingScore(post, agent.type);
  const keywordScore = this.keywordMatch(post.content, agent.tradition);
  const authorScore = this.getAuthorQuality(post.author);

  // Weighted average
  let score =
    semanticScore * 0.40 +
    trendingScore * 0.30 +
    keywordScore * 0.15 +
    authorScore * 0.10;

  // Quality gate: 50% penalty for banned phrases
  if (isBannedPhrase(post.content, agent.type)) {
    score *= 0.5;
  }

  return Math.min(score, 1.0);
}
